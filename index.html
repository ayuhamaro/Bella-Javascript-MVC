    </div>

    var page = new Jsmvc2('page');
    page.Ctrl.__init = function(){
        var self = this.o;
        $('#go').bind('click', function(){
            var rnd = Math.random();
            self.Model('ajax-test.html?rnd=' + rnd.toString(), 'go', {});
        });
    }
    page.CallBack.go = function(obj){
        var self = this.o;
        var rnd = Math.random();
        self.View('view-code.html?rnd=' + rnd.toString(), 'place', obj, function(html){});
    }
